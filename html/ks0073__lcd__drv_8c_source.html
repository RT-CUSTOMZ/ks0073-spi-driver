290 void DataConvert(uint8_t Data, KS0073_RWTypeDef RW, KS0073_RSTypeDef RS, KS0073_DataTypeDef * buffer)
291 {
292  //Five 1s as Start Bit
293  buffer->start = 0x1F;

298  if(RS == KS0073_RS_SET)
